def train_ensemble_classifier(X_train: np.ndarray, y_train: np.ndarray,
                             X_test: np.ndarray) -> np.ndarray :
    """
    Train a voting classifier of Random Forest and Gradient Boosting.
    Both use StandardScaler and handle  class imbalance via class_weight.
    This approach is more robust than neural networks for this dataset.
    """ 
    from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
    from sklearn.ensemble import VotingClassifier
     from sklearn.preprocessing import StandardScaler
    from sklearn.pipeline import Pipeline

    # Ensure all features are finite
     X_train = np.nan_to_num(X_train, nan=0, posinf=0 , neginf=0)
    X_test = np.nan_to_num(X_test,  nan=0, posinf=0, neginf=0)

    # Compute class weights with boosted emphasis on severe (class 2)
    n_cls = 3 
    counts = np.bincount(y_train, minlength=n_cls).astype(float )
    # handle case where a class is not in y_train
    counts = np.where(counts == 0 , 1.0, counts)
    class_weights = {i: 1.0 for i in  range(n_cls)}
    if counts.min() > 0:
        base_weights = { i: counts.sum() / (n_cls * counts[i]) for i in range(n_cls )}
        # Increase weight for severe class (2) by 1.5× to improve recall
        class_weights  = {
            0: base_weights[0],
            1: base_weights[1],
             2: 1.5 * base_weights[2]
        }

    # Random Forest pipeline — tuned for severe class  sensitivity
    rf = Pipeline([
        ('scaler', StandardScaler()),
        ('clf', RandomForestClassifier(
             n_estimators=300,
            max_depth=None,
            min_samples_leaf= 2,
            class_weight=class_weights,
            random_state=42,
            n_jobs=-1
        ))
    ])

    # Gradient Boosting pipeline — more robust against overfitting
    gb  = Pipeline([
        ('scaler', StandardScaler()),
        ('clf', GradientBoostingClassifier(
            n_estimators =300,          # more trees
            max_depth=3,               # reduce overfitting
            learning_rate =0.03,        # lower LR for stability
            subsample=0.8,
            random_state=42
        ))
    ])

    #  Soft voting ensemble
    ensemble = VotingClassifier(
        estimators=[('rf', rf), ('gb', gb)], 
        voting='soft'
    )

    ensemble.fit(X_train, y_train)
     return ensemble.predict(X_test)
