def train_ensemble_classifier(X, y, n_folds=5, random_state=42):
    """
    Trains an ensemble classifier using Random Forest and Gradient Boosting.
    Performs cross-validation for robust performance.
    """
    from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
    from sklearn.ensemble import VotingClassifier
    from sklearn.model_selection import StratifiedKFold, cross_validate

    n_samples, n_features = X.shape
    n_classes = len(np.unique(y))
    counts = np.bincount(y)

    # Define class weights (inverse frequency)
    class_weights = {i: counts.sum() / (n_classes * counts[i]) for i in range(n_classes)}

    # Define classifiers
    rf = RandomForestClassifier(n_estimators=300, min_samples_leaf=2, random_state=random_state)
    gb = GradientBoostingClassifier(n_estimators=200, max_depth=4, learning_rate=0.05, random_state=random_state)

    # Initialize soft voting ensemble
    ensemble = VotingClassifier(estimators=[('rf', rf), ('gb', gb)], voting='soft', weights=[0.5, 0.5])

    # Perform cross-validation
    cv = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=random_state)
    cv_scores = cross_validate(ensemble, X, y, scoring='f1_macro', cv=cv, fit_params={'rf__class_weight': class_weights, 'gb__class_weight': class_weights}, n_jobs=-1, return_estimator=True)
    best_idx = np.argmax(cv_scores['test_score'])
    best_model = cv_scores['estimator'][best_idx]
    return best_model

    class_weights = {i: counts.sum() / (n_cls * counts[i]) for i in range(n_cls)}

    # Random Forest pipeline
    rf = Pipeline([
        ('scaler', StandardScaler()),
        ('clf', RandomForestClassifier(
            n_estimators=300,
            min_samples_leaf=2,
            class_weight=class_weights,
            random_state=42,
            n_jobs=-1
        ))
    ])

    # Gradient Boosting pipeline
    gb = Pipeline([
        ('scaler', StandardScaler()),
        ('clf', GradientBoostingClassifier(
            n_estimators=200,
            max_depth=4,
            learning_rate=0.05,
            subsample=0.8,
            random_state=42
        ))
    ])

    # Soft voting ensemble
    ensemble = VotingClassifier(
        estimators=[('rf', rf), ('gb', gb)],
        voting='soft'
    )

    ensemble.fit(X_train, y_train)
    return ensemble.predict(X_test)
