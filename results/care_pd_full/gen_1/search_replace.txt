        # Batch process all windows for one walk
        # To avoid GPU OOM on very long walks, process in mini-batches
        walk_emb_list = []
        batch_size = 16  # MotionCLIP can handle batches
        for i in range(0, len(windows), batch_size):
            window_batch = windows[i:i + batch_size]
            x = torch.from_numpy(window_batch).float().to(DEVICE)

        all_window_embs = torch.cat(walk_emb_list, dim=0)    # (num_win, 512)
        walk_emb  = all_window_embs.mean(dim=0, keepdim=True) # (1, 512)
