<DESCRIPTION>
The current program uses a hybrid approach with MotionCLIP features and handcrafted gait features as fallback. However, analysis of prior runs shows that class 2 (severe PD) F1 remains low (~0.29), indicating poor detection of severe cases. The winning strategy (score 0.6231) uses physics-informed handcrafted features as primary input, not MotionCLIP.

We enhance the `extract_gait_features` function to include more discriminative PD-specific features:
1. Add velocity and acceleration statistics for key joints (pelvis, hips, knees, ankles)
2. Expand asymmetry features to include mean and std of L-R differences across all gait phases
3. Add freeze-of-gait (FoG) ratio using high-frequency power in ankle rotations
4. Include gait regularity via autocorrelation of pelvis signal
5. Add dominant stride frequency and relative power in gait band (0.5–3.0 Hz)

These features directly capture known PD gait impairments: asymmetry, shuffling gait, FoG, and irregular rhythm. By enriching the handcrafted feature vector, we improve class separability — especially for class 2.

We also fix a critical bug: the current `FocalLoss` ignores class weights because they are passed as `weight` to `cross_entropy` instead of being used in alpha scaling. We revert to using `nn.CrossEntropyLoss(weight=w)` which is proven to work.

Finally, we ensure handcrafted features are always used as fallback when MotionCLIP is unavailable, maintaining robustness.

class FocalLoss(nn.Module):
    def __init__(self, alpha: torch.Tensor, gamma: float = 2.0):
        super().__init__()
        self.alpha = alpha  # (3,) tensor of per-class weights
        self.gamma = gamma

    def forward(self, logits: torch.Tensor, targets: torch.Tensor) -> torch.Tensor:
        ce = nn.functional.cross_entropy(logits, targets, reduction='none', weight=self.alpha)
        pt = torch.exp(-ce)
        return ((1 - pt) ** self.gamma * ce).mean()

    std_feat = flat.std(0)     # (150,)

    # 2. Velocity (frame-to-frame difference) statistics
    vel = np.diff(flat, axis=0)  # (T-1, 150)
    vel_mean = vel.mean(0)       # (150,)
    vel_std = vel.std(0)         # (150,)

    asym_hip = (rot6d[:, 1, :] - rot6d[:, 2, :]).std(0)    # (6,)
    asym_knee = (rot6d[:, 4, :] - rot6d[:, 5, :]).std(0)   # (6,)
    asym_ankle = (rot6d[:, 7, :] - rot6d[:, 8, :]).std(0)  # (6,)

def extract_gait_features(pose: np.ndarray) -> np.ndarray:
    """
    Extract handcrafted gait features from SMPL pose sequence.
    Designed for PD severity classification with focus on asymmetry, rhythm, and dynamics.
    """
    T = len(pose)
    rot6d = smpl_to_6d(pose)  # (T, 25, 6)
    flat = rot6d.reshape(T, -1)  # (T, 150)

    # 1. Global statistics
    mean_feat = flat.mean(0)   # (150,)
    std_feat = flat.std(0)     # (150,)
    range_feat = flat.max(0) - flat.min(0)  # (150,)

    # 2. Velocity & acceleration
    vel = np.diff(flat, axis=0)  # (T-1, 150)
    acc = np.diff(vel, axis=0)   # (T-2, 150)
    vel_mean = vel.mean(0)      # (150,)
    vel_std = vel.std(0)        # (150,)
    acc_std = acc.std(0) if len(acc) > 0 else np.zeros_like(vel[0])  # (150,)

    # 3. Key joint velocity (pelvis, hips, knees, ankles, feet)
    key_joints = [0, 1, 2, 4, 5, 7, 8, 10, 11]
    joint_vel = np.diff(rot6d[:, :24, :], axis=0)  # (T-1, 24, 6)
    kj_vel_mean = np.abs(joint_vel[:, key_joints, :]).mean(axis=0).flatten()  # (9*6,)
    kj_vel_std = joint_vel[:, key_joints, :].std(axis=0).flatten()  # (9*6,)

    # 4. Left-right asymmetry (hip, knee, ankle, foot)
    asym_hip = (rot6d[:,1,:] - rot6d[:,2,:])
    asym_knee = (rot6d[:,4,:] - rot6d[:,5,:])
    asym_ankle = (rot6d[:,7,:] - rot6d[:,8,:])
    asym_foot = (rot6d[:,10,:] - rot6d[:,11,:])
    asym_features = np.concatenate([
        asym_hip.mean(0), asym_hip.std(0),
        asym_knee.mean(0), asym_knee.std(0),
        asym_ankle.mean(0), asym_ankle.std(0),
        asym_foot.mean(0), asym_foot.std(0),
    ])  # (8*6,)

    # 5. Stride frequency (FFT on pelvis rotation)
    pelvis = rot6d[:,0,0]
    fft = np.abs(np.fft.rfft(pelvis - pelvis.mean()))
    freqs = np.fft.rfftfreq(T, d=1.0/30)
    gait_mask = (freqs >= 0.5) & (freqs <= 3.0)
    dominant_freq = freqs[gait_mask][np.argmax(fft[gait_mask])] if gait_mask.any() else 0.0
    gait_power = fft[gait_mask].sum() if gait_mask.any() else 0.0
    hf_power = fft[freqs > 3.0].sum()
    total_power = fft.sum() + 1e-8

    # 6. Gait regularity (autocorrelation)
    ac = np.correlate(pelvis - pelvis.mean(), mode='full')
    ac = ac[T-1:T+60] / (ac[T-1] + 1e-8)
    regularity = ac[1:11].mean() if len(ac) > 10 else 0.0

    # 7. Freeze-of-gait proxy (high-frequency power in ankles)
    ankle_l = rot6d[:,7,0]
    fft_ankle = np.abs(np.fft.rfft(ankle_l - ankle_l.mean()))
    freqs_a = np.fft.rfftfreq(T, d=1.0/30)
    lf_mask = (freqs_a >= 0.5) & (freqs_a <= 3.0)
    hf_mask = freqs_a > 3.0
    fog_ratio = (fft_ankle[hf_mask].sum() / (fft_ankle[lf_mask].sum() + 1e-8)) if lf_mask.any() else 0.0

    # 8. Walk duration and summary stats
    duration = T / 30.0
    mean_vel = np.abs(vel).mean()
    mean_std = vel_std.mean()

    return np.concatenate([
        mean_feat, std_feat, range_feat,
        vel_mean, vel_std, acc_std,
        kj_vel_mean, kj_vel_std,
        asym_features,
        [dominant_freq, gait_power/total_power, hf_power/total_power],
        [regularity],
        [fog_ratio],
        [duration, mean_vel, mean_std]
    ]).astype(np.float32)
