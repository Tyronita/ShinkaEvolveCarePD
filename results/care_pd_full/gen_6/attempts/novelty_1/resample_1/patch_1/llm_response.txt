@torch.no_grad()
def extract_features(encoder: nn.Module, poses: list) -> np.ndarray:
    """
    Extract MotionCLIP embeddings for a list of variable-length pose arrays.

    Strategy: sliding-window aggregation — extract multiple 60-frame windows
    from each walk and average the embeddings. This captures more temporal
    information from longer walks (up to 3015 frames).

    Args:
        encoder: frozen MotionCLIPEncoder on DEVICE
        poses: list of (T_i, 72) float32 arrays
    Returns:
        (N, 512) numpy feature matrix
    """
    embs = []
    window = 60
    stride = 30  # 50% overlap

    for pose in poses:
        rot6d = smpl_to_6d(pose)   # (T, 25, 6)
        T = len(rot6d)

        # Collect windows
        windows = []
        if T <= window:
            # Pad short sequences
            seq = crop_pad_frames(rot6d, n=window)
            windows.append(seq)
        else:
            # Slide window with stride
            starts = list(range(0, T - window + 1, stride))
            if not starts:
                starts = [0]
            for s in starts:
                windows.append(rot6d[s:s + window])

        # Batch encode all windows
        batch = torch.from_numpy(
            np.stack(windows, axis=0)).float().to(DEVICE)  # (W, 60, 25, 6)
        feats = encoder(batch).cpu().numpy()               # (W, 512)

        # Mean pooling across windows
        embs.append(feats.mean(0, keepdims=True))          # (1, 512)

    return np.vstack(embs)                                 # (N, 512)

class ClassifierMLP(nn.Module):
    """
    MLP on top of frozen MotionCLIP embeddings with BatchNorm and residual.
    Mutable: depth, width, dropout, activation, batch norm, residual connections.
    """
    def __init__(self, in_dim: int = 512, n_classes: int = 3, dropout: float = 0.4):
        super().__init__()
        self.fc1 = nn.Linear(in_dim, 512)
        self.bn1 = nn.BatchNorm1d(512)
        self.fc2 = nn.Linear(512, 256)
        self.bn2 = nn.BatchNorm1d(256)
        self.fc3 = nn.Linear(256, 128)
        self.bn3 = nn.BatchNorm1d(128)
        self.fc4 = nn.Linear(128, n_classes)
        self.drop = nn.Dropout(dropout)
        self.act  = nn.GELU()
        # Residual projection from 512 → 256
        self.res1 = nn.Linear(512, 256)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        h1 = self.act(self.bn1(self.fc1(x)))
        h1 = self.drop(h1)
        h2 = self.act(self.bn2(self.fc2(h1)))
        h2 = h2 + self.res1(h1)   # residual
        h2 = self.drop(h2)
        h3 = self.act(self.bn3(self.fc3(h2)))
        h3 = self.drop(h3)
        return self.fc4(h3)


def train_classifier(feat_train: np.ndarray, y_train: np.ndarray,
                     feat_test:  np.ndarray,
                     epochs: int = 250, lr: float = 1e-3) -> np.ndarray:
    """
    Train ClassifierMLP on precomputed MotionCLIP features.

    Class weights computed from training label distribution to handle imbalance.
    Uses weighted CrossEntropyLoss + AdamW + cosine LR schedule.
    Strong class weights ensure class 2 (severe PD) is not ignored.

    Mutable: epochs, lr, batch_size, loss function, optimizer, schedule.
    """
    n_cls   = 3
    counts  = np.bincount(y_train, minlength=n_cls).astype(float)
    counts  = np.where(counts == 0, 1.0, counts)
    # Inverse frequency weights — class 2 (rare/severe) gets highest weight
    raw_w   = counts.sum() / (n_cls * counts)
    # Extra boost for class 2 (severe PD is clinically most important)
    raw_w[2] *= 1.5
    w       = torch.tensor(raw_w / raw_w.sum() * n_cls, dtype=torch.float32).to(DEVICE)

    clf  = ClassifierMLP(in_dim=feat_train.shape[1], n_classes=n_cls).to(DEVICE)
    opt  = optim.AdamW(clf.parameters(), lr=lr, weight_decay=1e-4)
    # Weighted CE with label smoothing for better calibration
    loss_fn = nn.CrossEntropyLoss(weight=w, label_smoothing=0.1)
    sch  = optim.lr_scheduler.CosineAnnealingLR(opt, T_max=epochs, eta_min=1e-5)

    X_tr = torch.from_numpy(feat_train.astype(np.float32)).to(DEVICE)
    y_tr = torch.tensor(y_train, dtype=torch.long).to(DEVICE)
    X_te = torch.from_numpy(feat_test.astype(np.float32)).to(DEVICE)

    N, bs = len(X_tr), 32
    clf.train()
    for ep in range(epochs):
        perm = torch.randperm(N)
        for s in range(0, N, bs):
            b = perm[s:s + bs]
            opt.zero_grad()
            loss_fn(clf(X_tr[b]), y_tr[b]).backward()
            nn.utils.clip_grad_norm_(clf.parameters(), 1.0)
            opt.step()
        sch.step()

    clf.eval()
    with torch.no_grad():
        return clf(X_te).argmax(1).cpu().numpy()
