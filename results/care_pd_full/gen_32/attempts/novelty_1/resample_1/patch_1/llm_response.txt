@torch.no_grad()
def extract_motionclip_std_features(encoder: nn.Module, poses: list) -> np.ndarray:
    """
    Extract std of MotionCLIP embeddings across windows (gait variability).
    """
    embs = []
    for pose in poses:
        rot6d = smpl_to_6d(pose)
        windows = get_sliding_windows(rot6d, window_size=60, step=30)
        x = torch.from_numpy(windows).float().to(DEVICE)
        window_embs = encoder(x)  # (N_w, 512)
        if window_embs.shape[0] > 1:
            std_emb = window_embs.std(0, keepdim=True).cpu().numpy()
        else:
            std_emb = np.zeros((1, 512), dtype=np.float32)
        embs.append(std_emb)
    return np.vstack(embs)
