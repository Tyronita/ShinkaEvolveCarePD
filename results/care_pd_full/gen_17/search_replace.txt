# ── Handcrafted gait feature extraction ──────────────────────────────────────

def extract_gait_features(pose: np.ndarray) -> np.ndarray:
    """
    Extract physics-informed gait features from raw SMPL pose.

    Args:
        pose: (T, 72) float32
    Returns:
        (D,) feature vector with gait-relevant statistics
    """
    T = len(pose)
    rot6d = smpl_to_6d(pose)  # (T, 25, 6)
    flat = rot6d.reshape(T, -1)  # (T, 150)

    # 1. Global statistics
    mean_feat = flat.mean(0)   # (150,)
    std_feat = flat.std(0)     # (150,)

    # 2. Velocity (frame-to-frame difference) statistics
    if T > 1:
        vel = np.diff(flat, axis=0)  # (T-1, 150)
        vel_mean = vel.mean(0)
        vel_std = vel.std(0)
    else:
        vel_mean = np.zeros(150, dtype=np.float32)
        vel_std = np.zeros(150, dtype=np.float32)

    # 3. Left-right asymmetry for key joint pairs
    # L_hip=1, R_hip=2, L_knee=4, R_knee=5, L_ankle=7, R_ankle=8
    # L_shoulder=16, R_shoulder=17
    asym_hip = np.abs(rot6d[:, 1, :] - rot6d[:, 2, :]).mean(0)      # (6,)
    asym_hip_std = np.abs(rot6d[:, 1, :] - rot6d[:, 2, :]).std(0)   # (6,)
    asym_knee = np.abs(rot6d[:, 4, :] - rot6d[:, 5, :]).mean(0)     # (6,)
    asym_knee_std = np.abs(rot6d[:, 4, :] - rot6d[:, 5, :]).std(0)  # (6,)
    asym_ankle = np.abs(rot6d[:, 7, :] - rot6d[:, 8, :]).mean(0)    # (6,)
    asym_ankle_std = np.abs(rot6d[:, 7, :] - rot6d[:, 8, :]).std(0) # (6,)

    # 4. Stride frequency via FFT on pelvis rotation
    pelvis = rot6d[:, 0, 0]  # (T,) first component of pelvis 6D
    if T > 10:
        fft_mag = np.abs(np.fft.rfft(pelvis - pelvis.mean()))
        freqs = np.fft.rfftfreq(T, d=1.0/30)  # 30 fps
        gait_mask = (freqs >= 0.5) & (freqs <= 3.0)
        if gait_mask.any() and fft_mag[gait_mask].max() > 0:
            stride_freq = freqs[gait_mask][np.argmax(fft_mag[gait_mask])]
            stride_power = fft_mag[gait_mask].max() / (fft_mag.sum() + 1e-8)
        else:
            stride_freq = 0.0
            stride_power = 0.0
    else:
        stride_freq = 0.0
        stride_power = 0.0

    # 5. Gait regularity: autocorrelation of pelvis
    if T > 30:
        pelvis_centered = pelvis - pelvis.mean()
        ac = np.correlate(pelvis_centered, pelvis_centered, mode='full')
        ac = ac[T-1:T+30] / (ac[T-1] + 1e-8)
        regularity = ac[1:min(11, len(ac))].mean() if len(ac) > 1 else 0.0
    else:
        regularity = 0.0

    # 6. Freeze-of-gait proxy: high-frequency power ratio in ankles
    if T > 10:
        ankle_l = rot6d[:, 7, 0]
        fft_ankle = np.abs(np.fft.rfft(ankle_l - ankle_l.mean()))
        freqs_a = np.fft.rfftfreq(T, d=1.0/30)
        hf_mask = freqs_a > 3.0
        lf_mask = (freqs_a >= 0.5) & (freqs_a <= 3.0)
        hf_sum = fft_ankle[hf_mask].sum() if hf_mask.any() else 0.0
        lf_sum = fft_ankle[lf_mask].sum() if lf_mask.any() else 1e-8
        fog_ratio = hf_sum / (lf_sum + 1e-8)
    else:
        fog_ratio = 0.0

    # 7. Range of motion for key joints
    rom_hip_l = (rot6d[:, 1, :].max(0) - rot6d[:, 1, :].min(0))   # (6,)
    rom_hip_r = (rot6d[:, 2, :].max(0) - rot6d[:, 2, :].min(0))   # (6,)
    rom_knee_l = (rot6d[:, 4, :].max(0) - rot6d[:, 4, :].min(0))  # (6,)
    rom_knee_r = (rot6d[:, 5, :].max(0) - rot6d[:, 5, :].min(0))  # (6,)

    # 8. Acceleration statistics
    if T > 2:
        acc = np.diff(flat, n=2, axis=0)  # (T-2, 150)
        acc_mean = acc.mean(0)
        acc_std = acc.std(0)
        # Jerkiness: mean absolute acceleration
        jerkiness = np.abs(acc).mean()
    else:
        acc_mean = np.zeros(150, dtype=np.float32)
        acc_std = np.zeros(150, dtype=np.float32)
        jerkiness = 0.0

    # 9. Trunk stability (spine joints: 3=spine1, 6=spine2, 9=spine3)
    trunk_std = rot6d[:, [3, 6, 9], :].std(0).flatten()  # (18,)

    # 10. Sequence length (normalized) as a feature
    seq_len = T / 300.0  # normalize by typical length

    return np.concatenate([
        mean_feat, std_feat, vel_mean, vel_std,
        asym_hip, asym_hip_std, asym_knee, asym_knee_std,
        asym_ankle, asym_ankle_std,
        rom_hip_l, rom_hip_r, rom_knee_l, rom_knee_r,
        acc_mean, acc_std, trunk_std,
        [stride_freq, stride_power, regularity, fog_ratio, jerkiness, seq_len]
    ])

def train_mlp_classifier(feat_train: np.ndarray, y_train: np.ndarray,
                         feat_test: np.ndarray,
                         epochs: int = 200, lr: float = 1e-3) -> np.ndarray:
    """
    Train ClassifierMLP with proper class-weighted loss and oversampling.
    Returns predicted logits (not argmax) for ensemble use.
    """
    n_cls = 3
    counts = np.bincount(y_train, minlength=n_cls).astype(float)
    counts = np.where(counts == 0, 1.0, counts)
    w = torch.tensor(counts.sum() / (n_cls * counts), dtype=torch.float32).to(DEVICE)

    # Oversample minority classes (especially class 2)
    idx_by_class = {c: np.where(y_train == c)[0] for c in range(n_cls)}
    max_count = max(len(v) for v in idx_by_class.values())
    aug_indices = []
    for c in range(n_cls):
        c_idx = idx_by_class[c]
        if len(c_idx) == 0:
            continue
        n_repeat = max(1, int(np.ceil(max_count / len(c_idx))))
        aug_indices.append(np.tile(c_idx, n_repeat)[:max_count])
    aug_indices = np.concatenate(aug_indices)
    np.random.shuffle(aug_indices)

    feat_train_aug = feat_train[aug_indices]
    y_train_aug = y_train[aug_indices]

    clf = ClassifierMLP(in_dim=feat_train.shape[1], n_classes=n_cls, dropout=0.3).to(DEVICE)
    opt = optim.AdamW(clf.parameters(), lr=lr, weight_decay=1e-3)
    # Use class-weighted CrossEntropyLoss (the critical fix!)
    loss_fn = nn.CrossEntropyLoss(weight=w)
    sch = optim.lr_scheduler.CosineAnnealingLR(opt, T_max=epochs)

    X_tr = torch.from_numpy(feat_train_aug.astype(np.float32)).to(DEVICE)
    y_tr = torch.tensor(y_train_aug, dtype=torch.long).to(DEVICE)
    X_te = torch.from_numpy(feat_test.astype(np.float32)).to(DEVICE)

    N, bs = len(X_tr), 64
    clf.train()
    for _ in range(epochs):
        perm = torch.randperm(N)
        for s in range(0, N, bs):
            b = perm[s:s + bs]
            opt.zero_grad()
            loss_fn(clf(X_tr[b]), y_tr[b]).backward()
            opt.step()
        sch.step()

    clf.eval()
    with torch.no_grad():
        logits = clf(X_te).cpu().numpy()  # (N_test, 3)
    return logits


def train_sklearn_classifiers(feat_train: np.ndarray, y_train: np.ndarray,
                               feat_test: np.ndarray) -> np.ndarray:
    """
    Train ensemble of sklearn classifiers with class_weight='balanced'.
    Returns averaged probability matrix (N_test, 3).
    """
    from sklearn.svm import SVC
    from sklearn.ensemble import GradientBoostingClassifier, RandomForestClassifier
    from sklearn.preprocessing import StandardScaler
    from sklearn.pipeline import Pipeline

    probas = []

    # SVM with balanced class weights
    try:
        svm_clf = Pipeline([
            ('scaler', StandardScaler()),
            ('svm', SVC(kernel='rbf', C=10.0, gamma='scale',
                       class_weight='balanced', probability=True,
                       decision_function_shape='ovr'))
        ])
        svm_clf.fit(feat_train, y_train)
        probas.append(svm_clf.predict_proba(feat_test))
    except Exception as e:
        print(f'[genome] SVM failed: {e}')

    # Random Forest with balanced class weights
    try:
        rf_clf = Pipeline([
            ('scaler', StandardScaler()),
            ('rf', RandomForestClassifier(
                n_estimators=500, max_depth=None,
                class_weight='balanced_subsample',
                min_samples_leaf=3, random_state=42))
        ])
        rf_clf.fit(feat_train, y_train)
        probas.append(rf_clf.predict_proba(feat_test))
    except Exception as e:
        print(f'[genome] RF failed: {e}')

    # Gradient Boosting (no native class_weight, use sample_weight)
    try:
        from sklearn.utils.class_weight import compute_sample_weight
        sw = compute_sample_weight('balanced', y_train)
        scaler = StandardScaler()
        X_scaled = scaler.fit_transform(feat_train)
        X_test_scaled = scaler.transform(feat_test)

        gb_clf = GradientBoostingClassifier(
            n_estimators=300, max_depth=4, learning_rate=0.05,
            subsample=0.8, min_samples_leaf=5, random_state=42)
        gb_clf.fit(X_scaled, y_train, sample_weight=sw)
        probas.append(gb_clf.predict_proba(X_test_scaled))
    except Exception as e:
        print(f'[genome] GB failed: {e}')

    if probas:
        return np.mean(probas, axis=0)  # (N_test, 3)
    else:
        # Fallback: uniform
        return np.ones((len(feat_test), 3)) / 3.0


def train_classifier(feat_train: np.ndarray, y_train: np.ndarray,
                     feat_test: np.ndarray,
                     epochs: int = 200, lr: float = 1e-3) -> np.ndarray:
    """
    Ensemble of MLP + sklearn classifiers with class 2 logit boosting.
    """
    # Get MLP logits
    mlp_logits = train_mlp_classifier(feat_train, y_train, feat_test, epochs=epochs, lr=lr)
    # Convert to probabilities
    mlp_probs = np.exp(mlp_logits - mlp_logits.max(1, keepdims=True))
    mlp_probs = mlp_probs / mlp_probs.sum(1, keepdims=True)

    # Get sklearn probabilities
    sklearn_probs = train_sklearn_classifiers(feat_train, y_train, feat_test)

    # Ensemble: weighted average (sklearn gets more weight for small data)
    ensemble_probs = 0.35 * mlp_probs + 0.65 * sklearn_probs

    # Boost class 2 logit to recover severe PD predictions
    ensemble_probs[:, 2] *= 1.3

    return ensemble_probs.argmax(1)
