This crossover combines the best elements from the top-performing programs:

1. **Feature extraction from the 0.63 program**: Uses the comprehensive handcrafted gait features including velocity/acceleration stats, L-R asymmetry, ROM, FFT stride frequency, gait regularity, and freeze-of-gait proxies. The simpler VotingClassifier ensemble achieved 0.63.

2. **Stacking meta-learner from the 0.61 program**: The stacking approach (GBC + SVC → LogisticRegression meta-learner) showed promise for class 2 detection (0.45 F1). We keep this but add RandomForest as a third base learner.

3. **Key improvements**:
   - Add RandomForest as a third base classifier in the stacking ensemble (from the 0.60 program which showed RF helps)
   - Use enhanced class weights with 2x boost for class 2 (severe PD) since all programs struggle most there
   - Add MotionCLIP multi-window features with batch processing for speed
   - Include the joint coordination/coupling features from the 0.61 program (cross-correlation, pelvis-ankle coupling)
   - Use GBC with sample_weight for proper class weighting (GBC doesn't support class_weight directly)
   - Batch MotionCLIP window processing to reduce GPU overhead and stay within time limits
   - Clean NaN/inf from features before scaling

The goal is to push class 2 F1 above 0.50 while maintaining class 0 and 1 performance, targeting macro-F1 > 0.65.

Task: 3-class classification (UPDRS_GAIT in {0, 1, 2}) on BMCLab dataset.

Fitness: macro-F1 (higher is better). Baseline: 0.565 (RandomForest).
Target: 0.68+ (CARE-PD LOSO SOTA), 0.74+ (MIDA SOTA).

Architecture overview:
  1. SMPL axis-angle (T, 72) → 6D rotation (T, 25, 6)   [inline, mutable]
  2. Crop/pad to 60 frames → (1, 60, 25, 6)              [inline, mutable]
  3. MotionCLIP Transformer encoder → mu (1, 512)         [inline, mutable]
  4. MLP classifier head → logits (1, 3)                  [inline, mutable]

Everything in the EVOLVE-BLOCK is mutable. Promising mutations:

# ── Rotation utilities ──────────────────────────────────────────────────────

def _aa_to_quat(aa: torch.Tensor) -> torch.Tensor:
    """Axis-angle (..., 3) → quaternion (..., 4) [w, x, y, z]."""
    angles = torch.norm(aa, p=2, dim=-1, keepdim=True).clamp(min=1e-8)
    half   = 0.5 * angles
    small  = angles.abs() < 1e-6
    s      = torch.where(small, 0.5 - angles * angles / 48, torch.sin(half) / angles)
    return torch.cat([torch.cos(half), aa * s], dim=-1)


def _quat_to_mat(q: torch.Tensor) -> torch.Tensor:
    """Quaternion (..., 4) [w, x, y, z] → rotation matrix (..., 3, 3)."""
    r, i, j, k = torch.unbind(q, -1)
    s = 2.0 / (q * q).sum(-1)
    o = torch.stack([
        1 - s*(j*j + k*k), s*(i*j - k*r),     s*(i*k + j*r),
        s*(i*j + k*r),     1 - s*(i*i + k*k), s*(j*k - i*r),
        s*(i*k - j*r),     s*(j*k + i*r),     1 - s*(i*i + j*j),
    ], dim=-1)
    return o.reshape(q.shape[:-1] + (3, 3))


def _mat_to_6d(mat: torch.Tensor) -> torch.Tensor:
    """Rotation matrix (..., 3, 3) → 6D representation (..., 6) [first 2 rows]."""
    return mat[..., :2, :].clone().reshape(*mat.shape[:-2], 6)


def smpl_to_6d(pose: np.ndarray) -> np.ndarray:
    """Convert SMPL axis-angle pose to 6D rotation + zero translation."""
    T  = len(pose)
    aa = torch.from_numpy(pose.reshape(T, 24, 3)).float()
    r6 = _mat_to_6d(_quat_to_mat(_aa_to_quat(aa))).numpy()
    return np.concatenate([r6, np.zeros((T, 1, 6), np.float32)], 1)


def get_sliding_windows(seq: np.ndarray, window_size: int = 60, step: int = 30) -> np.ndarray:
    """Generate sliding windows from sequence with specified step size."""
    T = len(seq)
    if T <= window_size:
        reps = (window_size + T - 1) // T
        seq = np.tile(seq, (reps,) + (1,) * (seq.ndim - 1))
        T = len(seq)
    starts = np.arange(0, T - window_size + 1, step)
    if len(starts) == 0:
        starts = np.array([0])
    windows = np.array([seq[s:s + window_size] for s in starts])
    return windows


def extract_gait_features(pose: np.ndarray) -> np.ndarray:
    """
    Extract comprehensive handcrafted gait features from SMPL pose sequence.
    Combines the best features from all prior programs.
    """
    T = len(pose)
    rot6d = smpl_to_6d(pose)  # (T, 25, 6)
    flat = rot6d.reshape(T, -1)  # (T, 150)

    features = []

    # 1. Global rotation statistics per joint (mean, std, percentiles, range)
    features.append(flat.mean(0))       # (150,)
    features.append(flat.std(0))        # (150,)
    features.append(np.percentile(flat, 25, axis=0))  # (150,)
    features.append(np.percentile(flat, 75, axis=0))  # (150,)

    # 2. Velocity (1st derivative) statistics
    vel = np.diff(flat, axis=0)  # (T-1, 150)
    features.append(vel.mean(0))
    features.append(vel.std(0))
    features.append(np.abs(vel).mean(0))

    # 3. Acceleration (2nd derivative) statistics — jerkiness proxy
    if T > 2:
        acc = np.diff(vel, axis=0)  # (T-2, 150)
        features.append(acc.mean(0))
        features.append(acc.std(0))
        features.append(np.abs(acc).mean(0))
    else:
        features.extend([np.zeros(150)] * 3)

    # 4. Key joint velocities (pelvis, hips, knees, ankles, feet)
    key_joints = [0, 1, 2, 4, 5, 7, 8, 10, 11]
    joint_vel = np.diff(rot6d[:, :24, :], axis=0)  # (T-1, 24, 6)
    kj_vel = joint_vel[:, key_joints, :]  # (T-1, 9, 6)
    features.append(np.abs(kj_vel).mean(axis=0).flatten())   # (54,)
    features.append(kj_vel.std(axis=0).flatten())             # (54,)

    # 5. Left-right asymmetry (KEY for PD — affected side shows reduced motion)
    lr_pairs = [(1, 2), (4, 5), (7, 8), (10, 11), (16, 17), (18, 19), (20, 21)]
    for l_idx, r_idx in lr_pairs:
        diff = rot6d[:, l_idx, :] - rot6d[:, r_idx, :]
        features.append(diff.mean(0))
        features.append(diff.std(0))
        features.append(np.abs(diff).mean(0))

    # 6. Joint range of motion (ROM) — reduced in PD
    for j in [0, 1, 2, 4, 5, 7, 8, 10, 11, 12, 16, 17, 18, 19]:
        joint_data = rot6d[:, j, :]
        rom = joint_data.max(0) - joint_data.min(0)
        features.append(rom)

    # 7. Stride frequency via FFT on key lower body joints
    for j in [0, 1, 2, 7, 8]:
        for comp in range(3):
            signal = rot6d[:, j, comp]
            signal = signal - signal.mean()
            if T > 10:
                fft_mag = np.abs(np.fft.rfft(signal))
                freqs = np.fft.rfftfreq(T, d=1.0/30)
                gait_mask = (freqs >= 0.5) & (freqs <= 3.0)
                hf_mask = freqs > 3.0
                total_power = fft_mag[1:].sum() + 1e-8
                if gait_mask.any():
                    gait_fft = fft_mag[gait_mask]
                    gait_freqs = freqs[gait_mask]
                    dom_freq = gait_freqs[np.argmax(gait_fft)]
                    gait_power_ratio = gait_fft.sum() / total_power
                    features.append(np.array([dom_freq, gait_power_ratio]))
                else:
                    features.append(np.zeros(2))
                hf_ratio = fft_mag[hf_mask].sum() / total_power if hf_mask.any() else 0.0
                features.append(np.array([hf_ratio]))
            else:
                features.append(np.zeros(3))

    # 8. Gait regularity via autocorrelation of pelvis
    pelvis_signal = rot6d[:, 0, 0]
    pelvis_signal = pelvis_signal - pelvis_signal.mean()
    if T > 30:
        ac = np.correlate(pelvis_signal, pelvis_signal, mode='full')
        ac = ac[T-1:] / (ac[T-1] + 1e-8)
        n_lags = min(60, len(ac))
        ac_feat = ac[:n_lags]
        if n_lags < 60:
            ac_feat = np.pad(ac_feat, (0, 60 - n_lags))
        features.append(ac_feat)
        features.append(np.array([ac[1:11].mean() if len(ac) > 10 else 0.0]))
    else:
        features.append(np.zeros(61))

    # 9. Freeze-of-gait proxy: high-freq power ratio for ankles
    for j in [7, 8]:
        signal = rot6d[:, j, 0] - rot6d[:, j, 0].mean()
        if T > 10:
            fft_mag = np.abs(np.fft.rfft(signal))
            freqs = np.fft.rfftfreq(T, d=1.0/30)
            hf = fft_mag[freqs > 3.0].sum()
            lf = fft_mag[(freqs >= 0.5) & (freqs <= 3.0)].sum() + 1e-8
            features.append(np.array([hf / lf]))
        else:
            features.append(np.zeros(1))

    # 10. Joint coordination — cross-correlation at key joint pairs
    coord_pairs = [(1, 2), (4, 5)]
    for l_idx, r_idx in coord_pairs:
        l_rot = rot6d[:, l_idx, :] - rot6d[:, l_idx, :].mean(axis=0)
        r_rot = rot6d[:, r_idx, :] - rot6d[:, r_idx, :].mean(axis=0)
        for comp in range(6):
            l_signal = l_rot[:, comp]
            r_signal = r_rot[:, comp]
            xcorr = np.correlate(l_signal, r_signal, 'full')
            l_auto = np.correlate(l_signal, l_signal, 'full')
            r_auto = np.correlate(r_signal, r_signal, 'full')
            den = np.sqrt(l_auto[len(l_auto)//2] * r_auto[len(r_auto)//2]) + 1e-8
            xcorr = xcorr / den
            center = len(xcorr) // 2
            lags = [center - 5, center, center + 5]
            lag_values = [xcorr[l] for l in lags if 0 <= l < len(xcorr)]
            features.append(np.array(lag_values))

    # 11. Pelvis-ankle coupling
    pelvis_rot = rot6d[:, 0, :]
    for ankle_idx in [7, 8]:
        ankle_rot = rot6d[:, ankle_idx, :]
        for comp in range(3):
            p_signal = pelvis_rot[:, comp] - pelvis_rot[:, comp].mean()
            a_signal = ankle_rot[:, comp] - ankle_rot[:, comp].mean()
            xcorr = np.correlate(p_signal, a_signal, 'full')
            p_auto = np.correlate(p_signal, p_signal, 'full')
            a_auto = np.correlate(a_signal, a_signal, 'full')
            den = np.sqrt(p_auto[len(p_auto)//2] * a_auto[len(a_auto)//2]) + 1e-8
            xcorr = xcorr / den
            center = len(xcorr) // 2
            lo = max(0, center - 10)
            hi = min(len(xcorr), center + 10)
            peak = np.argmax(np.abs(xcorr[lo:hi])) - (center - lo)
            features.append(np.array([peak]))

    # 12. Proximal-distal coupling ratio
    pelvis_var = rot6d[:, 0, :].var()
    ankle_var = rot6d[:, 7, :].var() + rot6d[:, 8, :].var() + 1e-8
    features.append(np.array([pelvis_var / ankle_var]))

    # 13. Walk duration and summary motion stats
    duration = T / 30.0
    mean_vel = np.abs(vel).mean()
    mean_vel_std = vel.std(0).mean()
    features.append(np.array([duration, np.log1p(T), mean_vel, mean_vel_std]))

    result = np.concatenate([f.ravel() for f in features]).astype(np.float32)
    return np.nan_to_num(result, nan=0.0, posinf=0.0, neginf=0.0)


# ── MotionCLIP encoder ──────────────────────────────────────────────────────

class _PosEnc(nn.Module):
    def __init__(self, d: int, drop: float = 0.1, max_len: int = 5000):
        super().__init__()
        self.drop = nn.Dropout(drop)
        pe = torch.zeros(max_len, d)
        pos = torch.arange(0, max_len).float().unsqueeze(1)
        div = torch.exp(torch.arange(0, d, 2).float() * (-np.log(10000.0) / d))
        pe[:, 0::2] = torch.sin(pos * div)
        pe[:, 1::2] = torch.cos(pos * div)
        self.register_buffer('pe', pe.unsqueeze(0).transpose(0, 1))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.drop(x + self.pe[:x.shape[0]])


class MotionCLIPEncoder(nn.Module):
    def __init__(self, njoints: int = 25, nfeats: int = 6,
                 latent_dim: int = 512, ff_size: int = 1024,
                 num_layers: int = 8, num_heads: int = 4,
                 dropout: float = 0.1):
        super().__init__()
        self.input_feats  = njoints * nfeats
        self.latent_dim   = latent_dim
        self.muQuery      = nn.Parameter(torch.randn(1, latent_dim))
        self.sigmaQuery   = nn.Parameter(torch.randn(1, latent_dim))
        self.embed        = nn.Linear(self.input_feats, latent_dim)
        self.pos_enc      = _PosEnc(latent_dim, dropout)
        layer = nn.TransformerEncoderLayer(
            d_model=latent_dim, nhead=num_heads,
            dim_feedforward=ff_size, dropout=dropout, activation='gelu')
        self.encoder      = nn.TransformerEncoder(layer, num_layers=num_layers)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, T, J, F = x.shape
        x = x.permute(1, 0, 2, 3).reshape(T, B, J * F)
        x = self.embed(x)
        y    = torch.zeros(B, dtype=torch.long, device=x.device)
        xseq = torch.cat([self.muQuery[y].unsqueeze(0),
                          self.sigmaQuery[y].unsqueeze(0), x], dim=0)
        xseq = self.pos_enc(xseq)
        mask = torch.ones(B, T + 2, dtype=torch.bool, device=x.device)
        out  = self.encoder(xseq, src_key_padding_mask=~mask)
        return out[0]


def _load_checkpoint(model: nn.Module, path: str) -> nn.Module:
    ckpt  = torch.load(path, map_location='cpu', weights_only=False)
    sd    = ckpt.get('state_dict', ckpt)
    msd   = model.state_dict()
    first = next(iter(msd))
    new   = collections.OrderedDict()
    for k, v in sd.items():
        if k.startswith('module.') and 'module.' not in first:
            k = k[7:]
        if k in msd:
            new[k] = v
    msd.update(new)
    model.load_state_dict(msd, strict=True)
    print(f'[genome] checkpoint: {len(new)}/{len(msd)} layers loaded from {os.path.basename(path)}')
    return model


def load_motionclip(care_pd_dir: str) -> nn.Module:
    if care_pd_dir is None:
        return None
    ckpt = os.path.join(care_pd_dir, 'assets', 'Pretrained_checkpoints',
                        'motionclip', 'motionclip_encoder_checkpoint_0100.pth.tar')
    if not os.path.isfile(ckpt):
        print(f'[genome] MotionCLIP checkpoint not found: {ckpt}')
        return None
    try:
        enc = MotionCLIPEncoder(
            njoints=25, nfeats=6, latent_dim=512,
            ff_size=1024, num_layers=8, num_heads=4).to(DEVICE)
        _load_checkpoint(enc, ckpt)
        enc.eval()
        for p in enc.parameters():
            p.requires_grad_(False)
        return enc
    except Exception as e:
        print(f'[genome] MotionCLIP load failed: {e}')
        return None


@torch.no_grad()
def extract_motionclip_features(encoder: nn.Module, poses: list) -> np.ndarray:
    """Extract MotionCLIP embeddings with batched multi-window mean+std pooling."""
    window_size = 60
    hop = 30
    batch_size = 32
    embs = []
    
    for pose in poses:
        rot6d = smpl_to_6d(pose)
        T = rot6d.shape[0]
        
        if T < window_size:
            reps = (window_size + T - 1) // T
            rot6d = np.tile(rot6d, (reps,) + (1,) * (rot6d.ndim - 1))[:window_size]
            T = window_size
        
        starts = list(range(0, T - window_size + 1, hop))
        if len(starts) == 0:
            starts = [0]
        
        windows = np.stack([rot6d[s:s + window_size] for s in starts])
        
        # Batch processing for efficiency
        all_window_embs = []
        for i in range(0, len(windows), batch_size):
            batch = torch.from_numpy(windows[i:i+batch_size]).float().to(DEVICE)
            batch_embs = encoder(batch)
            all_window_embs.append(batch_embs)
        
        window_embs = torch.cat(all_window_embs, dim=0)  # (N_w, 512)
        
        # Mean + std pooling for richer representation
        mean_emb = window_embs.mean(0)   # (512,)
        std_emb = window_embs.std(0)     # (512,)
        pooled = torch.cat([mean_emb, std_emb]).unsqueeze(0).cpu().numpy()  # (1, 1024)
        embs.append(pooled)
    
    return np.nan_to_num(np.vstack(embs), nan=0.0)


def train_and_predict(train_poses: list, y_train: np.ndarray,
                      test_poses: list, care_pd_dir: str = None) -> np.ndarray:
    """
    Hybrid pipeline: MotionCLIP (mean+std) + handcrafted gait features
    → 3-model stacking ensemble (GBC + SVC + RF → LR meta-learner)
    with boosted class 2 weights.
    """
    from sklearn.ensemble import GradientBoostingClassifier, RandomForestClassifier
    from sklearn.svm import SVC
    from sklearn.preprocessing import StandardScaler
    from sklearn.linear_model import LogisticRegression
    from sklearn.model_selection import StratifiedKFold

    enc = load_motionclip(care_pd_dir)

    # Extract handcrafted gait features
    print('[genome] Extracting handcrafted gait features...', flush=True)
    X_tr_gait = np.stack([extract_gait_features(p) for p in train_poses])
    X_te_gait = np.stack([extract_gait_features(p) for p in test_poses])
    print(f'[genome] Gait feature dim: {X_tr_gait.shape[1]}', flush=True)

    if enc is not None:
        print(f'[genome] Extracting MotionCLIP features (device={DEVICE})', flush=True)
        try:
            mc_tr = extract_motionclip_features(enc, train_poses)
            mc_te = extract_motionclip_features(enc, test_poses)
            X_tr = np.concatenate([X_tr_gait, mc_tr], axis=1)
            X_te = np.concatenate([X_te_gait, mc_te], axis=1)
            print(f'[genome] Combined feature dim: {X_tr.shape[1]}', flush=True)
        except Exception as e:
            print(f'[genome] MotionCLIP failed: {e}', flush=True)
            X_tr, X_te = X_tr_gait, X_te_gait
    else:
        print('[genome] Using handcrafted features only', flush=True)
        X_tr, X_te = X_tr_gait, X_te_gait

    # Compute class weights with extra boost for class 2
    n_cls = 3
    counts = np.bincount(y_train, minlength=n_cls).astype(float)
    counts = np.where(counts == 0, 1.0, counts)
    class_weights = {
        0: counts.sum() / (n_cls * counts[0]),
        1: counts.sum() / (n_cls * counts[1]),
        2: 2.0 * counts.sum() / (n_cls * counts[2])
    }
    sample_weight = np.array([class_weights[y] for y in y_train])
    
    print(f'[genome] Class weights: {class_weights}', flush=True)

    # Standardize features
    scaler = StandardScaler()
    X_tr_scaled = scaler.fit_transform(X_tr)
    X_te_scaled = scaler.transform(X_te)

    # Base classifiers
    gbc = GradientBoostingClassifier(
        n_estimators=300,
        max_depth=3,
        learning_rate=0.03,
        subsample=0.8,
        random_state=42
    )
    svc = SVC(
        kernel='rbf',
        C=10,
        gamma='scale',
        class_weight=class_weights,
        probability=True,
        random_state=42
    )
    rf = RandomForestClassifier(
        n_estimators=300,
        max_depth=None,
        min_samples_leaf=2,
        class_weight=class_weights,
        random_state=42,
        n_jobs=-1
    )

    # Stacking: out-of-fold predictions
    n_folds = 5
    skf = StratifiedKFold(n_splits=n_folds, shuffle=True, random_state=42)
    gbc_oof = np.zeros((X_tr.shape[0], 3))
    svc_oof = np.zeros((X_tr.shape[0], 3))
    rf_oof = np.zeros((X_tr.shape[0], 3))

    for train_idx, val_idx in skf.split(X_tr_scaled, y_train):
        X_f, y_f = X_tr_scaled[train_idx], y_train[train_idx]
        X_v = X_tr_scaled[val_idx]
        sw_f = sample_weight[train_idx]

        gbc.fit(X_f, y_f, sample_weight=sw_f)
        svc.fit(X_f, y_f)
        rf.fit(X_f, y_f)

        gbc_oof[val_idx] = gbc.predict_proba(X_v)
        svc_oof[val_idx] = svc.predict_proba(X_v)
        rf_oof[val_idx] = rf.predict_proba(X_v)

    # Meta-learner
    X_meta_train = np.hstack([gbc_oof, svc_oof, rf_oof])
    meta_learner = LogisticRegression(
        C=1.0,
        class_weight=class_weights,
        max_iter=1000,
        random_state=42
    )
    meta_learner.fit(X_meta_train, y_train)

    # Final predictions
    gbc.fit(X_tr_scaled, y_train, sample_weight=sample_weight)
    svc.fit(X_tr_scaled, y_train)
    rf.fit(X_tr_scaled, y_train)

    gbc_pred = gbc.predict_proba(X_te_scaled)
    svc_pred = svc.predict_proba(X_te_scaled)
    rf_pred = rf.predict_proba(X_te_scaled)

    X_meta_test = np.hstack([gbc_pred, svc_pred, rf_pred])
    return meta_learner.predict(X_meta_test)

def run_evaluation(data: dict, folds: dict, care_pd_dir: str = None) -> dict:
    """
    6-fold subject-level cross-validation harness. Fixed — not evolved.

    Prints per-class and per-fold F1 after each fold so the meta-LLM can
    read stdout and reason about which strategies improve class-2 (severe PD).
    """
    all_preds, all_labels = [], []
    per_fold_results = {}

    for fold_id, split in folds.items():
        def get_data(subjects):
            poses, labels = [], []
            for sub in subjects:
                if sub not in data:
                    continue
                for wd in data[sub].values():
                    poses.append(wd['pose'].astype(np.float32))
                    labels.append(int(wd['UPDRS_GAIT']))
            return poses, np.array(labels, dtype=np.int64)

        tr_poses, y_tr = get_data(split['train'])
        te_poses, y_te = get_data(split['eval'])

        if not te_poses or len(np.unique(y_tr)) < 2:
            continue

        preds = train_and_predict(tr_poses, y_tr, te_poses, care_pd_dir=care_pd_dir)
        all_preds.extend(preds.tolist())
        all_labels.extend(y_te.tolist())

        fold_f1 = float(f1_score(y_te, preds, average='macro', zero_division=0))
        pc_f1   = f1_score(y_te, preds, average=None, labels=[0,1,2], zero_division=0)
        per_fold_results[str(fold_id)] = {
            'macro_f1':      round(fold_f1, 4),
            'n_test_walks':  int(len(y_te)),
            'n_train_walks': int(len(y_tr)),
        }
        print(f'  fold {fold_id}: macro_f1={fold_f1:.4f}  '
              f'f1=[normal:{pc_f1[0]:.3f}, mild:{pc_f1[1]:.3f}, severe:{pc_f1[2]:.3f}]',
              flush=True)

    macro_f1     = float(f1_score(all_labels, all_preds, average='macro',  zero_division=0))
    per_class_f1 = f1_score(all_labels, all_preds, average=None, labels=[0,1,2], zero_division=0)

    print(f'\n[result] macro_f1={macro_f1:.4f}  '
          f'f1_class=[normal:{per_class_f1[0]:.4f}, mild:{per_class_f1[1]:.4f}, '
          f'severe:{per_class_f1[2]:.4f}]  device={DEVICE}', flush=True)

    return {
        'combined_score':   macro_f1,
        'all_preds':        all_preds,
        'all_labels':       all_labels,
        'per_fold_results': per_fold_results,
        'per_class_f1':     per_class_f1.tolist(),
    }
