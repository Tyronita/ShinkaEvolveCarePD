The root causes:
- Single 60-frame center-crop loses temporal information from longer walks
- FocalLoss with alpha=0.25 isn't aggressive enough for this imbalance
- No explicit class weighting in the loss
- The MLP classifier is too simple for this task

1. **Multi-window aggregation**: Extract features from multiple overlapping windows across the full walk, then aggregate with mean pooling - this captures more of the gait cycle
2. **Weighted CrossEntropy loss** with strong class weights for severe (class 2) instead of FocalLoss
3. **Better MLP** with batch normalization and stronger regularization
4. **Augmentation**: Random window sampling during training to see more of the walk

@torch.no_grad()
def extract_features(encoder: nn.Module, poses: list) -> np.ndarray:
    """
    Extract MotionCLIP embeddings for a list of variable-length pose arrays.

    Strategy: slide multiple 60-frame windows over the full walk, encode each,
    then average the embeddings. This captures the full gait pattern.

    Args:
        encoder: frozen MotionCLIPEncoder on DEVICE
        poses: list of (T_i, 72) float32 arrays
    Returns:
        (N, 512) numpy feature matrix
    """
    embs = []
    for pose in poses:
        rot6d = smpl_to_6d(pose)   # (T, 25, 6)
        T = len(rot6d)
        n = 60
        stride = 30  # 50% overlap

        # Collect windows
        windows = []
        if T <= n:
            # Short walk: pad to 60 frames
            windows.append(crop_pad_frames(rot6d, n=n))
        else:
            # Slide windows with stride
            starts = list(range(0, T - n + 1, stride))
            if not starts:
                starts = [0]
            # Always include center crop
            center_start = (T - n) // 2
            all_starts = sorted(set(starts + [center_start]))
            for s in all_starts:
                windows.append(rot6d[s:s + n])

        # Encode all windows and average
        window_embs = []
        for w in windows:
            x = torch.from_numpy(w).float().unsqueeze(0).to(DEVICE)  # (1, 60, 25, 6)
            window_embs.append(encoder(x).cpu().numpy())              # (1, 512)
        avg_emb = np.mean(window_embs, axis=0)  # (1, 512)
        embs.append(avg_emb)

    return np.vstack(embs)   # (N, 512)

# ── Classifier head ──────────────────────────────────────────────────────────

class ClassifierMLP(nn.Module):
    """
    MLP with LayerNorm and residual connection on top of frozen MotionCLIP embeddings.
    Mutable: depth, width, dropout, activation, batch norm, residual connections.
    """
    def __init__(self, in_dim: int = 512, n_classes: int = 3, dropout: float = 0.4):
        super().__init__()
        hidden = 256
        self.fc1 = nn.Linear(in_dim, hidden)
        self.ln1 = nn.LayerNorm(hidden)
        self.act1 = nn.GELU()
        self.drop1 = nn.Dropout(dropout)

        self.fc2 = nn.Linear(hidden, hidden)
        self.ln2 = nn.LayerNorm(hidden)
        self.act2 = nn.GELU()
        self.drop2 = nn.Dropout(dropout)

        self.fc3 = nn.Linear(hidden, 128)
        self.ln3 = nn.LayerNorm(128)
        self.act3 = nn.GELU()
        self.drop3 = nn.Dropout(0.2)

        self.out = nn.Linear(128, n_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        h = self.drop1(self.act1(self.ln1(self.fc1(x))))
        # Residual block
        h2 = self.drop2(self.act2(self.ln2(self.fc2(h))))
        h = h + h2  # residual connection
        h = self.drop3(self.act3(self.ln3(self.fc3(h))))
        return self.out(h)


def train_classifier(feat_train: np.ndarray, y_train: np.ndarray,
                     feat_test:  np.ndarray,
                     epochs: int = 300, lr: float = 5e-4) -> np.ndarray:
    """
    Train ClassifierMLP on precomputed MotionCLIP features.

    Class weights computed from training label distribution to handle imbalance.
    Uses weighted CrossEntropyLoss (stronger than FocalLoss for severe class) +
    AdamW + cosine LR schedule with warm restarts.

    Mutable: epochs, lr, batch_size, loss function, optimizer, schedule.
    """
    n_cls   = 3
    counts  = np.bincount(y_train, minlength=n_cls).astype(float)
    counts  = np.where(counts == 0, 1.0, counts)
    # Class weights: inverse frequency, normalized — boost class 2 (severe)
    w_raw   = counts.sum() / (n_cls * counts)
    # Extra boost for class 2 (severe PD) which is most clinically important
    w_raw[2] *= 2.0
    w       = torch.tensor(w_raw / w_raw.sum() * n_cls, dtype=torch.float32).to(DEVICE)

    clf  = ClassifierMLP(in_dim=feat_train.shape[1], n_classes=n_cls).to(DEVICE)
    opt  = optim.AdamW(clf.parameters(), lr=lr, weight_decay=1e-4)
    # Weighted CrossEntropy: directly addresses class imbalance
    loss_fn = nn.CrossEntropyLoss(weight=w, label_smoothing=0.05)
    sch  = optim.lr_scheduler.CosineAnnealingWarmRestarts(opt, T_0=100, T_mult=2)

    X_tr = torch.from_numpy(feat_train.astype(np.float32)).to(DEVICE)
    y_tr = torch.tensor(y_train, dtype=torch.long).to(DEVICE)
    X_te = torch.from_numpy(feat_test.astype(np.float32)).to(DEVICE)

    N, bs = len(X_tr), 32
    clf.train()
    for _ in range(epochs):
        perm = torch.randperm(N)
        for s in range(0, N, bs):
            b = perm[s:s + bs]
            opt.zero_grad()
            loss_fn(clf(X_tr[b]), y_tr[b]).backward()
            nn.utils.clip_grad_norm_(clf.parameters(), max_norm=1.0)
            opt.step()
        sch.step()

    clf.eval()
    with torch.no_grad():
        return clf(X_te).argmax(1).cpu().numpy()
