def _aa_to_quat(aa):
    """
    Convert axis-angle representation to quaternion.
    """
    rot = R.from_rotvec(aa).as_quat()
    return rot

def _quat_to_mat(q):
    """
    Convert quaternion to rotation matrix.
    """
    rot = R.from_quat(q).as_matrix()
    return rot

def _mat_to_6d(mat):
    """
    Convert rotation matrix to 6D vector representation.
    """
    a = mat[:, 0]
    b = mat[:, 1]
    return np.concatenate([a, b])

def smpl_to_6d(pose: np.ndarray) -> np.ndarray:
    """
    Convert SMPL pose representation (axis-angle) to 6D vector representation.

    pose: (N, 72)
    """
    num_frames = pose.shape[0]
    result = np.zeros((num_frames, 24, 6))  # 24 joints, 6D representation

    for i in range(24):
        aa = pose[:, i * 3:(i + 1) * 3]
        quat = _aa_to_quat(aa)
        rot_mat = _quat_to_mat(quat)
        result[:, i] = _mat_to_6d(rot_mat)
    return result.reshape(num_frames, -1)  # Flatten the last two dimensions

def extract_gait_features(pose: np.ndarray) -> np.ndarray:

  # Calculate velocities

  # Calculate average velocity across the body (all joints)
  avg_velocity = np.mean(np.linalg.norm(velocities, axis=1), axis=1)

  # Calculate asymmetry (e.g., difference between left and right leg movements)
